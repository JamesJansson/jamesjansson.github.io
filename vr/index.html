Demo 3: Add a building instead of ball.
function Building(x, y){
this.x = x;
this.y = y;
this.width = 20;
this.height = 30 + 10 * Math.random();
}

Building.prototype.addToScene = function (scene){
var material = new THREE.MeshPhongMaterial({
color: 0x00ff00,
specular: 0xffffff,
shininess: 20,
shading: THREE.FlatShading
}); var geometry = new THREE.BoxGeometry( this.width, this.height, this.width );
var cube = new THREE.Mesh( geometry, material );
scene.add( cube );
};
